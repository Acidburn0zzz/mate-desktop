typedef struct MateRRCrtc MateRRCrtc;

enum MateRRRotation

typedef enum
{
    MATE_RR_ROTATION_0 = (1 << 0),
    MATE_RR_ROTATION_90 = (1 << 1),
    MATE_RR_ROTATION_180 = (1 << 2),
    MATE_RR_ROTATION_270 = (1 << 3),
    MATE_RR_REFLECT_X = (1 << 4),
    MATE_RR_REFLECT_Y = (1 << 5)
} MateRRRotation;
